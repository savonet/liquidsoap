Module Srt.Log

type msg = {
  1. level : int;
  2. file : string;
  3. line : int;
  4. area : string;
  5. message : string;
}
type level = [
  1. | `Critical
  2. | `Error
  3. | `Warning
  4. | `Notice
  5. | `Debug
]
val setloglevel : level -> unit
val set_handler : (msg -> unit) -> unit
val clear_handler : unit -> unit
